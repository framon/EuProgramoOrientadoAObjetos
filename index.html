Afinal, o que é OO?

Paradigma de Programação

  • Fornece e determina a visão que o programador possui sobre a estruturação e execução do programa.
  • Diferencia-se de Linguagens de Programação

Na qualidade de método de modelagem, é tida como a melhor estratégia para se eliminar o "gap semântico", dificuldade recorrente no processo de modelar o mundo real do domínio do problema em um conjunto de componentes de software que seja o mais fiel na sua representação deste domínio.

Facilitaria a comunicação do profissional modelador e do usuário da área alvo, na medida em que a correlação da simbologia e conceitos abstratos do mundo real e da ferramenta de modelagem (conceitos, terminologia, símbolos, grafismo e estratégias) fosse a mais óbvia, natural e exata possível.

public class PedidoService {

  public void calcularTotal(Pedido pedido) {
    if (pedido == null) {
      throw new IllegalArgumentException("pedido must not be null");
    }

    BigDecimal total = BigDecimal.ZERO;
    List<PedidoItem> itens = pedido.getItens();
    
    for (PedidoItem pedidoItem : itens) {
      int quantidade = pedidoItem.getQuantidade();
      BigDecimal preco = pedidoItem.getPreco();
      BigDecimal itemTotal = preco.multiply(new BigDecimal(quantidade));
      total = total.add(itemTotal);
    }

    pedido.setTotal(total);
    
  }
}

public class ItemSaida extends ItemNotaMovimentacao implements Cloneable {
  
  public void adicionarItemInspecionadoRelacionado(ItemInspecionado itemInspecionado, Long quantidade) {
    AssertUtils.assertNotNull(itemInspecionado, "msg.erro.nnmeentrada.obrigatorio");        
    AssertUtils.assertNotNull(quantidade, "msg.erro.quantidade.obrigatorio");
    AssertUtils.assertExpression(quantidade > 0, "msg.erro.quantidade.maiorquezero");

    ItemRelacionado itemRelacionado = this.obterItemInspecionadoRelacionado(itemInspecionado);
    if (itemRelacionado != null) {
      itemRelacionado.setQuantidade(itemRelacionado.getQuantidade() + quantidade);
            
    } else {
      itemRelacionado = new ItemRelacionado();
      itemRelacionado.setItemSaida(this);
      itemRelacionado.setQuantidade(quantidade);
      itemRelacionado.setItemInspecionado(itemInspecionado);
    }
    itemInspecionado.debitar(quantidade);
        
    this.setQtdMaterial(this.getQtdMaterial() + quantidade);
    this.itensRelacionados.add(itemRelacionado);
    this.itensRelacionadosCarregados = true;
    this.firePropertyChange(PROP_ITENS_RELACIONADOS, null, null);
  }
}

public class ItemInspecionado extends AbstractIdBean<Long> {
  public void debitar(Long quantidade) {
    AssertUtils.assertExpression(saldo >= quantidade, 
      ConstantesI18N.ERRO_SALDO_NEGATIVO, getMaterial().getDescricao());
        
    Object old = this.saldo;
    this.saldo = saldo - quantidade;
    firePropertyChange(PROP_SALDO, old, this.saldo);
  }
    
  public void creditar(Long quantidade) {
    AssertUtils.assertExpression(quantidade + this.saldo <= this.quantidade, 
      ConstantesI18N.ERRO_SALDO_SUPERIOR_QTD);
        
    Object old = this.saldo;
    this.saldo = saldo + quantidade;
    firePropertyChange(PROP_SALDO, old, this.saldo);
  }
    
  public Long getSaldo() {
    return saldo;
  }

  public void setSaldo(Long saldo) {
    Object old = this.saldo;
    this.saldo = saldo;
    firePropertyChange("saldo", old, this.saldo);
  }
}
